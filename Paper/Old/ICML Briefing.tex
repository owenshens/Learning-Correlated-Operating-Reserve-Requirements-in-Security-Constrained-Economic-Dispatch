\section{The Bilevel Optimization (No Context)}
%===============================================================================

\subsection{Problem Statement}

\begin{problem}[Co-optimization of Geometry and Decision]
\begin{equation}\label{eq:bilevel}
\boxed{
\min_{\btheta,\, \rho > 0} V(\btheta, \rho) \quad \text{s.t.} \quad \Prob(\bu \in \cU_{\btheta,\rho}) \ge \tau
}
\end{equation}
where:
\begin{itemize}[leftmargin=2em, itemsep=2pt]
\item $\btheta \in \Theta$ = \textbf{shape} parameter (e.g., Cholesky factor $\bL$)
\item $\rho > 0$ = \textbf{size} parameter (radius)
\item $\tau \in (0,1)$ = target \textbf{coverage} probability
\item $V(\btheta,\rho)$ = robust value function (defined below)
\end{itemize}
\end{problem}

\begin{definition}[Robust Value Function]
\begin{equation}\label{eq:value-function}
V(\btheta,\rho) := \min_{\bx \in \cX} f(\bx) \quad \text{s.t.} \quad a_j(\bx) + \sigma_{\cU_{\btheta,\rho}}(\bw_j) \le b_j, \quad \forall j \in [m]
\end{equation}
where $\sigma_{\cU}(\bw) := \sup_{\bu \in \cU} \ip{\bw}{\bu}$ is the \textbf{support function}.
\end{definition}


\subsection{How We Solve It}

\paragraph{Step 1: Shape Optimization using Gradients.}
We optimize the \textbf{normalized} shape parameter $\btheta$ (with $\text{tr}(\bL\bL^\top) = d$ for ellipsoids) via gradient descent:
\begin{equation}\label{eq:envelope}
\boxed{
\nabla_{\btheta} V = \sum_{j=1}^m \mu_j^* \nabla_{\btheta} \sigma_j(\btheta,\rho)
}
\end{equation}
where $\mu_j^* \ge 0$ are optimal dual multipliers. 

\paragraph{Step 2: Conformal Calibration for Size.}
Given shape $\btheta$, set $\rho$ as the $\tau$-quantile of gauge scores on calibration data:
\begin{equation}
\hat{\rho}_\tau = \text{quantile}_\tau\left(\{s_{\btheta}(\bu_i)\}_{i=1}^{n_{\text{cal}}}\right), \quad s_{\btheta}(\bu) := \inf\{\rho' > 0 : \bu \in \cU_{\btheta,\rho'}\}
\end{equation}
This yields \textbf{distribution-free coverage}: $\Prob(\bu_{\text{new}} \in \cU_{\btheta,\hat{\rho}_\tau}) \ge \tau$.

\section{The Dispatch Problem with Context}
%===============================================================================

\subsection{Economic Dispatch Formulation}

\begin{problem}[Robust Economic Dispatch]
Given context $\bxi$ (load forecasts, renewable forecasts, temporal features, LMP-based costs):
\begin{align}
\min_{\bg, \br \ge 0} \quad & \sum_i c_i^g(\bxi) g_i + \sum_i c_i^r(\bxi) r_i \label{eq:dispatch-obj}\\
\text{s.t.} \quad & \sum_i g_i = D(\bxi) \quad \text{(power balance)} \label{eq:balance}\\
& g_i + r_i \le \bar{g}_i \quad \text{(capacity)} \label{eq:capacity}\\
& r_i \ge \sigma_{\cU_{\bL,\rho}}(\bw_i) = \rho \|\bL^\top \bw_i\|_2 \quad \text{(reserve requirement)} \label{eq:reserve}
\end{align}
\end{problem}

\paragraph{Variables and context-dependent parameters:}
\begin{itemize}[leftmargin=2em, itemsep=2pt]
\item $\bg \in \R^d$ = generation dispatch (MW), $\br \in \R^d$ = reserve allocation (MW)
\item $D(\bxi)$ = total demand (from forecast)
\item $c_i^g(\bxi), c_i^r(\bxi)$ = generation and reserve costs (LMP-based, vary with system state)
\end{itemize}



\subsection{The Learning Problem}

\begin{problem}[Contextual Geometry Learning]
Learn encoder $\bL_\phi : \bxi \mapsto \bL$ and calibrate $\rho$ to minimize expected dispatch cost subject to coverage:
\begin{equation}\label{eq:learning}
\boxed{
\min_{\phi,\, \rho > 0} \E\left[V(\bL_\phi(\bxi), \rho; \bxi)\right] \quad \text{s.t.} \quad \Prob(\bu \in \cU_{\bL_\phi(\bxi),\rho}) \ge \tau
}
\end{equation}
\end{problem}

\paragraph{Training loss:}
\begin{equation}
\mathcal{L}(\phi) = \E\left[\text{Dispatch Cost}(\bg^*, \br^*; \bxi)\right] = \E\left[\sum_i c_i^g g_i^* + \sum_i c_i^r r_i^*\right]
\end{equation}

\paragraph{Gradient (via envelope theorem):}
\begin{equation}
\nabla_{\bL} V = \sum_i \mu_i^* \nabla_{\bL} \sigma_i = \sum_i \mu_i^* \cdot \rho \cdot \frac{\bL^\top \bw_i \bw_i^\top}{\|\bL^\top \bw_i\|_2}
\end{equation}
where $\mu_i^* \ge 0$ is the dual on the reserve constraint for zone $i$.

\section{What Is Learned and What Is Covered}
%===============================================================================

\subsection{Context $\bxi$ and Uncertainty $\bu$}

\begin{center}
\small
\begin{tabular}{lrl}
\toprule
\textbf{Context $\bxi \in \R^p$} & \textbf{Dim} & \textbf{Components} \\
\midrule
Weather & 5 & temperature, wind speed, cloud cover, pressure gradient, humidity \\
Temporal & 9 & hour/day/month (sin/cos), weekday, peak, ramp indicators \\
System state & 4 & normalized load, load fraction, renewable penetration, horizon \\
Reserve costs & $d$ & zone-wise LMP-based prices \\
Lagged errors & $k \times d$ & $\bu_{t-1}, \ldots, \bu_{t-k}$ \\
\midrule
\textbf{Total (IEEE 118-bus)} & $912$ & $k=24$ hours $\times$ $(28 + d)$, with $d=10$ zones \\
\bottomrule
\end{tabular}
\end{center}

\paragraph{Uncertainty $\bu \in \R^d$:} Net load forecast error = (actual $-$ forecast) for (demand $-$ renewable). With $d=10$ zones, $u_i$ is the forecast error in zone $i$ (MW).


\subsection{What Is Covered}

\paragraph{Data structure:} Each time $t$ yields one pair $(\bxi_t, \bu_t)$. Dataset: $\{(\bxi_1, \bu_1), \ldots, (\bxi_n, \bu_n)\}$.

\paragraph{Coverage (marginal):}
\begin{equation}\label{eq:coverage}
\boxed{
\text{Coverage} := \Prob_{(\bxi,\bu) \sim P}\left(\bu \in \cU_{\bL_\phi(\bxi),\rho}\right) \ge \tau
}
\end{equation}
This is \textbf{marginal} over the joint $(\bxi, \bu)$, not conditional $\Prob(\bu \in \cU \mid \bxi = \bxi_0)$.

\paragraph{How (conformal calibration):}
\begin{enumerate}[leftmargin=2em, itemsep=1pt]
\item Compute gauge scores: $s_i = \|\bL_\phi(\bxi_i)^{-1} \bu_i\|_2$ on calibration set
\item Set $\hat{\rho}_\tau = \text{quantile}_\tau(\{s_1, \ldots, s_{n_{\text{cal}}}\})$
\item The set covers \textbf{uncertainty realizations} (for example, 95 percent) of $\bu$ (forecast errors), not predictions
\end{enumerate}
